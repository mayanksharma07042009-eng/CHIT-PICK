            <button class="btn" id="addOption">Add Option</button>

        </div>

    <script>
        class ChitpickingBox {
            constructor() {
                this.options = [];
                this.isSpinning = false;
                this.init();
            }

            init() {
                this.box = document.getElementById('box');
                this.spinBtn = document.getElementById('spinBtn');
                this.optionInput = document.getElementById('optionInput');
                this.optionsList = document.getElementById('optionsList');
                this.result = document.getElementById('result');
                
                this.addDefaultOptions();
                this.bindEvents();
                this.updateBoxFaces();
            }

            addDefaultOptions() {
                const defaults = ['🍕 Pizza', '🍔 Burger', '🌮 Tacos', '🍜 Ramen', '🍰 Cake', '☕ Coffee'];
                defaults.forEach(option => {
                    this.options.push(option);
                });
                this.updateOptionsList();
                this.updateBoxFaces();
                this.spinBtn.disabled = false;
            }

            bindEvents() {
                document.getElementById('addOption').addEventListener('click', () => this.addOption());
                document.getElementById('spinBtn').addEventListener('click', () => this.spin());
                this.optionInput.addEventListener('keypress', (e) => {
                    if (e.key === 'Enter') this.addOption();
                });
            }

            addOption() {
                const text = this.optionInput.value.trim();
                if (text && !this.options.includes(text) && this.options.length < 6) {
                    this.options.push(text);
                    this.optionInput.value = '';
                    this.updateOptionsList();
                    this.updateBoxFaces();
                }
            }

            removeOption(index) {
                this.options.splice(index, 1);
                this.updateOptionsList();
                this.updateBoxFaces();
                if (this.options.length === 0) {
                    this.spinBtn.disabled = true;
                }
            }

            updateOptionsList() {
                this.optionsList.innerHTML = '';
                this.options.forEach((option, index) => {
                    const li = document.createElement('li');
                    li.className = 'option-item';
                    li.innerHTML = `
                        ${option}
                        <button class="delete-btn" onclick="chitBox.removeOption(${index})">&times;</button>
                    `;
                    this.optionsList.appendChild(li);
                });
            }

            updateBoxFaces() {
                const faces = document.querySelectorAll('.box-face');
                faces.forEach((face, index) => {
                    if (this.options[index]) {
                        face.textContent = this.options[index];
                        face.style.background = `linear-gradient(45deg, 
                            hsl(${Math.random() * 360}, 70%, 60%), 
                            hsl(${Math.random() * 360}, 70%, 50%))`;
                    } else {
                        face.textContent = `Option ${index + 1}`;
                        face.style.background = 'linear-gradient(45deg, #666, #888)';
                    }
                });
            }

            spin() {
                if (this.isSpinning || this.options.length === 0) return;

                this.isSpinning = true;
                this.spinBtn.disabled = true;
                this.result.textContent = '';
                this.optionInput.disabled = true;

                // Random spin duration and final position
                const duration = 3500 + Math.random() * 1500;
                const spins = 10 + Math.random() * 5;
                const finalSide = Math.floor(Math.random() * this.options.length);
                
                this.box.classList.add('spin');
                this.box.style.setProperty('--spin-duration', `${duration}ms`);

                setTimeout(() => {
                    this.box.classList.remove('spin');
                    this.showResult(finalSide);
                    this.isSpinning = false;
                    this.spinBtn.disabled = false;
                    this.optionInput.disabled = false;
                }, duration);
            }

            showResult(side) {
                const resultText = this.options[side];
                this.result.textContent = `🎉 ${resultText} 🎉`;
                this.result.style.background = 'linear-gradient(45deg, #00d4aa, #00a085)';
                
                // Confetti effect
                this.createConfetti();
            }

            createConfetti() {
                for (let i = 0; i < 50; i++) {
                    const confetti = document.createElement('div');
                    confetti.style.cssText = `
                        position: fixed;
                        width: 10px;
                        height: 10px;
                        background: hsl(${Math.random() * 360}, 70%, 60%);
                        left: ${Math.random() * 100}vw;
                        top: -10px;
                        animation: confetti-fall 3s linear forwards;
                        z-index: 1000;
                        pointer-events: none;
                    `;
                    confetti.style.setProperty('--delay', `${Math.random() * 3}s`);
                    document.body.appendChild(confetti);

                    setTimeout(() => confetti.remove(), 3000);
                }
            }
        }

        // Global reference for delete buttons
        let chitBox;

        // Add confetti animation
        const confettiStyle = document.createElement('style');
        confettiStyle.textContent = `
            @keyframes confetti-fall {
                to {
                    transform: translateY(100vh) rotate(720deg);
                    opacity: 0;
                }
            }
        `;
        document.head.appendChild(confettiStyle);

        // Initialize when page loads
        window.addEventListener('load', () => {
            chitBox = new ChitpickingBox();
        });

        // Make chitBox global for delete buttons
        window.chitBox = chitBox;
    </script>
